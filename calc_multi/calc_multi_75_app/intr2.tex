import mpl_toolkits.mplot3d as a3
import matplotlib.colors as colors
import pylab as pl
import numpy as np

ax = a3.Axes3D(pl.figure())

def plot_tri3d(corners, axx):
   tri = a3.art3d.Poly3DCollection([corners])
   tri.set_edgecolor('k')
   tri.set_color('red')
   tri.set_alpha(0.2)
   axx.add_collection3d(tri)

vtx1 = np.array([[0,5,0],
                 [8,0,0 ],
                 [0,0,0]])

vtx2 = np.array([[6,8,3],
                 [6,8,-2],
                 [6,-4,-2]])

vtx3 = vtx2 + 3

plot_tri3d(vtx1,ax)
plot_tri3d(vtx2,ax)
plot_tri3d(vtx3,ax)

ax.view_init(elev=5, azim=250)

ax.set_xlim(-5,8)
ax.set_ylim(-5,8)
ax.set_zlim(-5,8)
plt.savefig('/tmp/out2.jpg')
