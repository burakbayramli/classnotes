import scipy.sparse as sps
from scipy.io import mmwrite, mmread
A = sps.lil_matrix((5,5))
A[0,1] = 0.5
A[1,0] = 0.5
A[1,2] = 1.0
A[2,1] = 1.0
A[2,3] = 0.33
A[3,2] = 0.33
A[0,4] = 0.5
A[4,0] = 0.5
mmwrite('./felzclust/simple.mtx',A)
